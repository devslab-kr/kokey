Converter — both directions at once

Paste anything below — toEn auto-detects all 8 layouts.

Install

# npm
npm install @devslab/kokey

# CDN — no build step
<script src="https://cdn.jsdelivr.net/npm/@devslab/kokey/dist/kokey.global.js"></script>

Usage

import { koToEn, enToKo, register, toEn, observe } from '@devslab/kokey'
import { ru } from '@devslab/kokey/ru'   // /uk /he /el /th /ar /ka

koToEn('안녕')   // 'dkssud'
enToKo('dkssud') // '안녕'

register(ru)
toEn('привет안녕') // 'ghbdtndkssud'

// Vue / React: v-model & controlled inputs
// <KokeyInput v-model="v" mode="ru" />

observe() // bind every <input data-kokey="ko|ru|en|…">
